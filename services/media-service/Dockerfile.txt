# Development Dockerfile for Media Service
FROM node:22-alpine

# Install system dependencies for media processing and development tools
RUN apk add --no-cache \
    ffmpeg \
    imagemagick \
    vips-dev \
    curl \
    netcat-openbsd \
    python3 \
    make \
    g++ \
    libc6-compat && \
    npm install -g nodemon && \
    addgroup -g 1001 -S nodejs && \
    adduser -S mediaservice -u 1001 && \
    mkdir -p /app/logs /app/temp /app/uploads && \
    chown -R mediaservice:nodejs /app

# Set working directory
WORKDIR /app

# Copy package files
COPY services/media-service/package*.json ./

# Install all dependencies (including dev dependencies)
RUN npm install

# Copy source code
COPY shared/ ./shared/
COPY services/media-service/src/ ./src/

# Switch to non-root user
USER mediaservice

# Expose port and debugging port
EXPOSE 3002 9230

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
    CMD curl -f http://localhost:3002/health || exit 1

# Start with nodemon for development
CMD ["nodemon", "--inspect=0.0.0.0:9230", "src/app.js"]
